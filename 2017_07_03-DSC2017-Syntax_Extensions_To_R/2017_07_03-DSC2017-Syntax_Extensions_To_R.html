Type Annotations

Example - Python 3

def greeting(name: str) -> str:
    hi: str = 'Hello '
    return hi + name

print(greeting("Jim"))

Proposed Syntax

# install.packages("types")
suppressPackageStartupMessages(library(types))
greeting <- function(name = ? str) {
  hi <- "Hello " ? str
  paste0(hi, name)
} ? str
greeting("Jim")
## [1] "Hello Jim"

String interpolation

Example - Bash

value="foo"
echo "value = $value"
## value = foo

Proposed Syntax

#install.package("glue")
g <- glue::glue
value <- "foo"
g("value = {value}")
## value = foo
g("1 + 1 = {1 + 1}")
## 1 + 1 = 2

Vs sprintf()

  • No explicit typing
  • Implicitly documented
  • Maintain locality

Variable unpacking

Example - Ruby

data = ['a', 'b', 'c']
foo, bar, baz  = data
puts(foo, bar, baz)
## a
## b
## c
`%<-%` <- function(lhs, rhs) {
  lhs <- substitute(lhs)[-1L]
  for (i in seq_along(lhs)) {
    assign(as.character(lhs[[i]]), rhs[[i]], envir = parent.frame())
  }
}
data <- c("a", "b", "c")
c(foo, bar, baz) %<-% data
foo; bar; baz
## [1] "a"
## [1] "b"
## [1] "c"

Proposed syntax

foo, bar, baz <- data

foreach and Generator functions

Example - Python

import warnings
def yrange(n):
    i = 0
    while i < n:
        yield i
        i += 1

y = yrange(3)
print(y)
print(y.next())
print(y.next())
print(y.next())
try:
    print(y.next())
except StopIteration as e:
    print("Iterator done")

# Within a loop
for i in yrange(3):
    print(i)
## <generator object yrange at 0x10a576aa0>
## 0
## 1
## 2
## Iterator done
## 0
## 1
## 2
yrange <- function(n) {
  i <- 1
  function() {
    if (i <= n) {
      i <<- i + 1
      return(i - 1)
    }
    return(FALSE)
  }
}
y = yrange(3)
y()
## [1] 1
y()
## [1] 2
y()
## [1] 3
y()
## [1] FALSE
y <- yrange(3)
while (i <- y()) {
  print(i)
}
## [1] 1
## [1] 2
## [1] 3

Proposed Syntax

foreach (i in yrange(3)) {
  print(i)
}

#> [1] 1
#> [1] 2
#> [1] 3
